\begin{document}
\begin{tikzpicture}
    \begin{axis}[
        axis lines=middle,
        xmin=0, xmax=2.5,
        ymin=0, ymax=2.5,
        xtick={0,1,2},
        ytick={0,1,2},
        xlabel={$x_1$},
        ylabel={$x_2$},
        width=7cm,

        \addplot[->, thick,blue] coordinates {(0,0) (1,1)};

        \node[gray!70!black,above left,font=\small] at (axis cs:1.5,1.5) {$\lambda = 2$};
